Wann ist das Projekt fertig?

            
select round((
  select sum(verbleibend)
  from
  (
    select
    pnr, arbeitsstundenschätzung - arbeitsstunden as verbleibend
    from projekt
    join
    (
      select pnr, sum(stunden) as arbeitsstunden 
      from arbeitstag 
      group by pnr
    ) c
    using (pnr)
  ) a
)
/
(
  select avg(wochenstunden) as avg
  from
  (
    select sum(stunden) as wochenstunden
    from arbeitstag
    group by date_part('week',tag),date_part('year',tag)
  ) b
),2) as verbleibende_wochen;

Views!

...eine virtuelle Tabelle ohne eigene Datensätze

Gearbeitete Wochenstunden

            
CREATE OR REPLACE VIEW wochenstunden AS
SELECT
SUM(stunden) AS wochenstunden
FROM arbeitstag
GROUP BY date_part('week',tag),date_part('year',tag)

Stunden pro Projekt

            
CREATE OR REPLACE VIEW projektarbeitsstunden AS
SELECT pnr, sum(stunden) AS arbeitsstunden
FROM arbeitstag GROUP BY pnr;

Verbleibende Studen

            
CREATE OR REPLACE VIEW verbleibend AS
SELECT
SUM(arbeitsstundenschätzung - arbeitsstunden) as verbleibend, pnr  
FROM projekt
JOIN projektarbeitsstunden USING (pnr)
GROUP BY pnr

Wochen bis Fertigstellung

            
SELECT ROUND(
(SELECT SUM(verbleibend) FROM verbleibend)
/
(SELECT avg(wochenstunden) FROM wochenstunden)

Projekt MUSS einen Supervisor haben

Infos zu Vererbung: Postgres Dokumentation 5.10.1

Supervisor Trigger


CREATE OR REPLACE FUNCTION require_supervisor() RETURNS TRIGGER 
LANGUAGE plpgsql AS $$ DECLARE
is_supervisor bool;
project_has_supervisor bool;
BEGIN
    SELECT position = 'supervisor' INTO is_supervisor
    FROM mitarbeiter
    WHERE mnr = NEW.mnr;

    IF is_supervisor THEN
	    RETURN NEW;
    END IF;
    
    SELECT count(*)>0 INTO project_has_supervisor
	  FROM arbeitetbei
	  JOIN mitarbeiter USING (mnr)
 	  WHERE position = 'supervisor'
	  AND pnr = NEW.pnr;

    IF NOT project_has_supervisor
    THEN
        RAISE EXCEPTION 'Ein Projekt muss einen Supervisor haben';
    END IF;
    RETURN NEW;
END;
$$ ;
